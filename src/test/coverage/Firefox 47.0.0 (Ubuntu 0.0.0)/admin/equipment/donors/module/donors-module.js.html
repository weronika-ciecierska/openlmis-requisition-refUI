angular.module('donor', ['openlmis', 'ngGrid', 'ui.bootstrap.modal', 'ui.bootstrap.dialog']).
    config(['$routeProvider', function ($routeProvider) {
      $routeProvider.
        when('/list', {controller: DonorListController, templateUrl: 'partials/list.html'}).
        when('/create-donor', {controller: DonorController, templateUrl: 'partials/create.html',resolve:DonorController.resolve}).
        when('/edit/:donorId', {controller: DonorController, templateUrl: 'partials/create.html',resolve:DonorController.resolve}).
        otherwise({redirectTo: '/list'});
    }]).directive('onKeyup', function () {
      return function (scope, elm, attrs) {
        elm.bind("keyup", function () {
          scope.$apply(attrs.onKeyup);
        });
      };
    })
    .directive('select2Blur', function () {
      return function (scope, elm, attrs) {
        angular.element("body").on('mousedown', function (e) {
          $('.select2-dropdown-open').each(function () {
            if (!$(this).hasClass('select2-container-active')) {
              $(this).data("select2").blur();
            }
          });
        });
      };
    })
    .run(function ($rootScope, AuthorizationService) {
      $rootScope.donorSelected = "selected";
      AuthorizationService.preAuthorize('MANAGE_DONOR');
    })
    .service('sharedSpace',function(){
        var donationCount;
 
        return {
            getCountOfDonations: function() {
                return donationCount;
            },
 
            setCountOfDonations: function(value){
                donationCount = value;
            }
        };
    });
